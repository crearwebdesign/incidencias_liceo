CREATE TABLE public.familiar (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  ci_representante text NOT NULL UNIQUE,
  nombres_representante text NOT NULL,
  apellidos_representante text NOT NULL,
  telefono_1 text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT familiar_pkey PRIMARY KEY (id)
);
CREATE TABLE public.alumnos (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  ni smallint,
  ci text NOT NULL UNIQUE,
  nombres text NOT NULL,
  apellidos text NOT NULL,
  nacionalidad text,
  lugar_nacimiento text,
  fecha_nacimiento date,
  sexo text,
  estatus text DEFAULT 'REGULAR'::text,
  anio_actual smallint,
  seccion_actual text,
  periodo_origen text,
  anio_origen text,
  seccion_origen text,
  fecha_retiro date,
  observacion_retiro text,
  observaciones text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  representante_id bigint,
  padre_id bigint,
  pensum character varying,
  nlistaactual smallint,
  entidadfederal character varying,
  materiaPendient1 text,
  materiaPendient2 text,
  CONSTRAINT alumnos_pkey PRIMARY KEY (id),
  CONSTRAINT alumnos_representante_id_fkey FOREIGN KEY (representante_id) REFERENCES public.familiar(id),
  CONSTRAINT alumnos_padre_id_fkey FOREIGN KEY (padre_id) REFERENCES public.padres(id)
);

CREATE TABLE public.materias (

CREATE TABLE public.usuarios (
  id uuid NOT NULL,
  email text NOT NULL UNIQUE,
  nombres text NOT NULL,
  apellidos text NOT NULL,
  rol text NOT NULL CHECK (rol = ANY (ARRAY['admin'::text, 'director'::text, 'docente'::text, 'secretaria'::text, 'representante'::text])),
  telefono text,
  activo boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT usuarios_pkey PRIMARY KEY (id),
  CONSTRAINT usuarios_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
